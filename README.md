# iConfig

```shell
# install
git clone git@github.com:sunzhenkai/iconfig.git ~/.config/iconfig
cd ~/.config/iconfig  make all

# config
cd ~/.config/iconfig/init && git submodule update --init
./activate

# install programs
ii homebrew -i
ii homebrew -c

ii asdf -i
ii asdf -c

ii zsh -c

## config programs
# config oh-my-zsh
vim ~/.zshrc
ZSH_CUSTOME=~/.config/zsh/oh-my-zsh

```
